# ============================================
# MARIYAE E-COMMERCE - ENVIRONMENT VARIABLES
# ============================================
# 
# IMPORTANT: Create a file named .env.local in the root directory
# and copy the content below, then update with your actual credentials
# 
# ============================================

# MongoDB Database Configuration
# Update with your new MongoDB connection string
MONGODB_URI=mongodb+srv://your_username:your_password@your_cluster.mongodb.net/mariyae?retryWrites=true&w=majority

# MongoDB Database Name (optional, defaults to 'mariyae' if not specified)
MONGODB_DB=mariyae

# ============================================
# Cloudinary Configuration
# All media files will be saved in folder: mariyae-com/
# ============================================
# 
# Update with your new Cloudinary credentials
# Get these from: https://console.cloudinary.com/settings/api-keys
#
CLOUDINARY_CLOUD_NAME=your_cloud_name
CLOUDINARY_API_KEY=your_api_key
CLOUDINARY_API_SECRET=your_api_secret

# ============================================
# Authentication & Security
# ============================================

# JWT Secret for user authentication
# Generate a secure random string (at least 32 characters)
# Example: openssl rand -base64 32
JWT_SECRET=your_jwt_secret_key_here_make_it_long_and_secure

# NextAuth Configuration
NEXTAUTH_URL=http://localhost:3000
NEXTAUTH_SECRET=your_nextauth_secret_here

# ============================================
# NOTES:
# ============================================
# 
# 1. Cloudinary Folder Structure:
#    - All data saved in: mariyae-com/
#    - Products: mariyae-com/products/
#    - Categories: mariyae-com/categories/{category}/
#    - Banners: mariyae-com/banners/
#    - Thumbnails: mariyae-com/thumbnails/
#
# 2. Database:
#    - Database name: mariyae (or as specified in MONGODB_DB)
#    - Collections: products, orders, users, main_categories, sub_categories
